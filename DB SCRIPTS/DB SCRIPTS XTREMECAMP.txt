CREATE TABLE CAMPSITES (
ID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
NAME VARCHAR(100) NOT NULL, 
DESCRIPTION VARCHAR(300) NOT NULL, 
FOUNDATIONDATE DATE NOT NULL, 
MAXNUMRESERVATIONSPERDAY INT NOT NULL,
MAXNUMDAYSINCLUDEDINDATERANGE INT NOT NULL,
MINDAYSBEFOREINITDATEBOOKING INT NOT NULL,
MAXDAYSBEFOREINITDATEBOOKING INT NOT NULL, 
STATUS VARCHAR(2) NOT NULL, 
CREATEDBY VARCHAR(50) NOT NULL, 
CREATEDAT DATE NOT NULL, 
MODIFIEDBY VARCHAR(50), 
MODIFIEDAT DATE);

CREATE TABLE CUSTOMERS (
ID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
FIRSTNAME VARCHAR(100) NOT NULL, 
LASTNAME VARCHAR(100) NOT NULL, 
EMAIL VARCHAR(100) NOT NULL,
STATUS VARCHAR(2) NOT NULL, 
CREATEDBY VARCHAR(50) NOT NULL, 
CREATEDAT DATE NOT NULL, 
MODIFIEDBY VARCHAR(50), 
MODIFIEDAT DATE);

CREATE TABLE BOOKINGS (
ID BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
IDCAMPSITE BIGINT NOT NULL, 
IDCUSTOMER BIGINT NOT NULL, 
BOOKINGCODE VARCHAR(64) NOT NULL,
BOOKINGINITDATE DATE NOT NULL,
BOOKINGENDDATE DATE NOT NULL,
STATUS VARCHAR(2) NOT NULL, 
CREATEDBY VARCHAR(50) NOT NULL, 
CREATEDAT DATE NOT NULL, 
MODIFIEDBY VARCHAR(50), 
MODIFIEDAT DATE,
OPTLOCK BIGINT);
